signup
curl -X POST http://localhost:5000/api/users/register -H "Content-Type: application/json" -d "{\"name\":\"John Doe\",\"email\":\"johndoe@example.com\",\"password\":\"password123\",\"role\":\"customer\"}"

login
curl -X POST http://localhost:5000/api/users/login -H "Content-Type: application/json" -d "{\"email\":\"johndoe@example.com\",\"password\":\"password123\"}"

get profile
curl -X GET http://localhost:5000/api/users/profile -H "Authorization: Bearer YOUR_JWT_TOKEN"

get all items in the marketplace
curl -X GET http://localhost:5000/api/items

get item by id
curl -X GET http://localhost:5000/api/items/item_id

add item to cart
curl -X POST http://localhost:5000/api/items/cart -H "Authorization: Bearer YOUR_JWT_TOKEN" -H "Content-Type: application/json" -d "{\"itemId\":\"item_id\"}"

get all cart items
curl -X GET http://localhost:5000/api/items/cart -H "Authorization: Bearer YOUR_JWT_TOKEN"

get seller story
curl -X GET http://localhost:5000/api/items/story/seller_id

remove item from cart
curl -X DELETE http://localhost:5000/api/items/cart -H "Authorization: Bearer JWT" -H "Content-Type: application/json" -d "{\"itemId\":\"id\"}"
